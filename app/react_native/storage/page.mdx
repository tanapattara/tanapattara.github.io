# Local Storage ใน React Native

## การเก็บข้อมูลลงเครื่อง

แอปพลิเคชันมักต้องการเก็บข้อมูลบางอย่างไว้บนเครื่อง เพื่อให้ข้อมูลยังคงอยู่แม้จะปิดแอปแล้วเปิดใหม่ เช่น การตั้งค่า ประวัติ หรือ Token สำหรับล็อกอิน

React Native มี 2 ตัวเลือกหลัก:
- **AsyncStorage** — เก็บข้อมูลทั่วไป (ไม่เข้ารหัส) เช่น Preference, Cache
- **SecureStore** (expo-secure-store) — เก็บข้อมูลสำคัญ (เข้ารหัส) เช่น Token, รหัสผ่าน

## AsyncStorage

AsyncStorage เป็นระบบเก็บข้อมูล Key-Value บนเครื่อง รองรับเฉพาะ String ดังนั้นหากต้องการเก็บ Object ต้องแปลงเป็น JSON String ก่อน

```bash
# ติดตั้ง AsyncStorage
npx expo install @react-native-async-storage/async-storage
```

```tsx
import AsyncStorage from '@react-native-async-storage/async-storage';
import { useState, useEffect } from 'react';
import { View, Text, TextInput, Button, StyleSheet } from 'react-native';

const STORAGE_KEY = '@user_name'; // กำหนด Key เป็น Constant เพื่อป้องกัน Typo

export default function StorageDemo(): React.JSX.Element {
  const [name, setName] = useState<string>('');
  const [savedName, setSavedName] = useState<string>('');

  // โหลดข้อมูลที่บันทึกไว้เมื่อเปิดแอป
  useEffect(() => {
    loadName();
  }, []);

  const loadName = async () => {
    try {
      const value = await AsyncStorage.getItem(STORAGE_KEY);
      if (value !== null) {
        setSavedName(value); // มีข้อมูลอยู่แล้ว ให้โหลดมาแสดง
      }
    } catch (err) {
      console.error('โหลดข้อมูลไม่สำเร็จ:', err);
    }
  };

  const saveName = async () => {
    try {
      await AsyncStorage.setItem(STORAGE_KEY, name); // บันทึกข้อมูล
      setSavedName(name);
      setName('');
    } catch (err) {
      console.error('บันทึกข้อมูลไม่สำเร็จ:', err);
    }
  };

  const deleteName = async (): Promise<void> => {
    try {
      await AsyncStorage.removeItem(STORAGE_KEY); // ลบข้อมูล
      setSavedName('');
    } catch (err) {
      console.error('ลบข้อมูลไม่สำเร็จ:', err);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.label}>ชื่อที่บันทึกไว้: {savedName || '(ยังไม่มี)'}</Text>
      <TextInput
        style={styles.input}
        value={name}
        onChangeText={setName}
        placeholder="กรอกชื่อของคุณ"
      />
      <Button title="บันทึก" onPress={saveName} />
      <Button title="ลบข้อมูล" onPress={deleteName} color="#FF3B30" />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { padding: 20, gap: 12 },
  label: { fontSize: 16, color: '#333' },
  input: {
    borderWidth: 1,
    borderColor: '#ccc',
    borderRadius: 8,
    padding: 10,
    fontSize: 15,
  },
});
```

## เก็บ Object ด้วย JSON

เนื่องจาก AsyncStorage รองรับเฉพาะ String จึงต้องใช้ `JSON.stringify` ก่อนบันทึก และ `JSON.parse` เมื่อโหลดออกมา

```ts
import AsyncStorage from '@react-native-async-storage/async-storage';

const SETTINGS_KEY = '@app_settings';

interface AppSettings {
  theme: string;
  language: string;
  notifications: boolean;
}

// บันทึก Object
const saveSettings = async (settings: AppSettings): Promise<void> => {
  try {
    const jsonValue = JSON.stringify(settings); // แปลง Object → String
    await AsyncStorage.setItem(SETTINGS_KEY, jsonValue);
  } catch (err) {
    console.error('บันทึก settings ไม่สำเร็จ:', err);
  }
};

// โหลด Object
const loadSettings = async (): Promise<AppSettings | null> => {
  try {
    const jsonValue = await AsyncStorage.getItem(SETTINGS_KEY);
    return jsonValue != null ? (JSON.parse(jsonValue) as AppSettings) : null; // แปลง String → Object
  } catch (err) {
    console.error('โหลด settings ไม่สำเร็จ:', err);
    return null;
  }
};

// ตัวอย่างการใช้งาน
await saveSettings({ theme: 'dark', language: 'th', notifications: true });
const settings = await loadSettings();
if (settings) {
  console.log(settings.theme); // "dark"
}
```

## SecureStore (เก็บข้อมูลสำคัญ)

สำหรับข้อมูลที่ต้องการความปลอดภัยสูง เช่น Token, รหัสผ่าน ควรใช้ `expo-secure-store` ซึ่งเข้ารหัสข้อมูลด้วยระบบ Keychain (iOS) หรือ Keystore (Android)

```bash
# ติดตั้ง SecureStore
npx expo install expo-secure-store
```

```ts
import * as SecureStore from 'expo-secure-store';

const TOKEN_KEY = 'auth_token';

// บันทึก Token หลังจาก Login สำเร็จ
const saveToken = async (token: string): Promise<void> => {
  await SecureStore.setItemAsync(TOKEN_KEY, token);
};

// อ่าน Token เพื่อส่งใน API Request
const getToken = async (): Promise<string | null> => {
  const token = await SecureStore.getItemAsync(TOKEN_KEY);
  return token; // คืนค่า null ถ้าไม่มี
};

// ลบ Token เมื่อ Logout
const removeToken = async (): Promise<void> => {
  await SecureStore.deleteItemAsync(TOKEN_KEY);
};

// ตัวอย่างการใช้งานครบวงจร
async function handleLogin(username: string, password: string): Promise<void> {
  const response = await fetch('https://api.example.com/login', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ username, password }),
  });
  const { token } = await response.json() as { token: string };
  await saveToken(token); // เก็บ Token อย่างปลอดภัย
}
```
